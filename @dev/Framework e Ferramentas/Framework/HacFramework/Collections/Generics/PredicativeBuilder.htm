//Predicative utilizado em generics, exemplo nos métodos Exists e FindAll.
	    
        static void Main(string[] args)
        {            
            //add some employees:
            List employeeList = new List();
            employeeList.Add(new Employee("John", "Doe", 55000));
            employeeList.Add(new Employee("Larry", "Jones", 65000));
            employeeList.Add(new Employee("Wayne", "Smith", 25000));
            employeeList.Add(new Employee("Sally", "Johnson", 35000));
            employeeList.Add(new Employee("Maggie", "Brown", 60000));
            employeeList.Add(new Employee("David", "Brown", 80000));
           
            //output them
            Console.WriteLine("All Employees: \n");
            employeeList.ForEach(new Action(Employee.Print));
            Console.WriteLine("\n\n\n");
            
            //Verifica se existe algum "john" (sem casesensitive)
            bool bExistsJohn = employeeList.Exists(PredicateBuilder.Build("FirstName = john", false));
            
            //select out the Employee's whose last name is Brown (case sensitive):
            List matchesLastName = employeeList.FindAll(PredicateBuilder.Build("LastName = Brown"));
            
            //output the matches
            Console.WriteLine("The Brown's:\n");
            
            matchesLastName.ForEach(new Action(Employee.Print));
            Console.WriteLine("\n\n\n");
            
            //everybody making $50000 or more
            List matches50K = employeeList.FindAll(PredicateBuilder.Build("Salary >= 50000"));

            //output the matches
            Console.WriteLine("People making 50k or more:\n");
            matches50K.ForEach(new Action(Employee.Print));
            Console.WriteLine("\n\n\n");
            Console.ReadLine();
